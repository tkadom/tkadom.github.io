What is a Microcontroller vs. a Microprocessor?

What is an Arduino?

  • Physical computing platform
  • USB programmable
  • Large Community
  • Multiple platforms
  • Based on ATmega328 - 8bit
  • 32k Memory
  • PWM, I2C, SPI
  • 14 Digital I/O ports (6 PWM outputs)
  • 6 analog inputs

Find a good signal level for low light.. 

int photoRPin = 0; 
int lightLevel;

void setup() {
 Serial.begin(9600);
}

void loop(){
 lightLevel=analogRead(photoRPin);
 if ( lightLevel < 600 )
     Serial.println("SIGNAL");
 
 delay(50);
}

#include < Servo.h >

Servo myservo;
int photoRPin = 0; 
int lightLevel;

void setup() {
 Serial.begin(9600);
 myservo.attach(9);
}

void loop(){
 lightLevel=analogRead(photoRPin);
 myservo.write(45);
 
 if ( lightLevel < 600 )
 {
     Serial.println("SIGNAL");
     myservo.write(135);
 }
 
 delay(50);
}
